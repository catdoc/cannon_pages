179  CANNON_FORCE_INLINE Quaternion& mult(const Quaternion& q, Quaternion& target) const {
180 #ifdef CANNON_NEON_64
181  float32x4_t vQ1 = mVec128;
182  float32x4_t vQ2 = q.get128();
183  float32x4_t A0, A1, B1, A2, B2, A3, B3;
184  float32x2_t vQ1zx, vQ2wx, vQ1yz, vQ2zx, vQ2yz, vQ2xz;
185 
186  {
187  float32x2x2_t tmp;
188  tmp = vtrn_f32(vget_high_f32(vQ1), vget_low_f32(vQ1)); // {z x}, {w y}
189  vQ1zx = tmp.val[0];
190 
191  tmp = vtrn_f32(vget_high_f32(vQ2), vget_low_f32(vQ2)); // {z x}, {w y}
192  vQ2zx = tmp.val[0];
193  }
194  vQ2wx = vext_f32(vget_high_f32(vQ2), vget_low_f32(vQ2), 1);
195 
196  vQ1yz = vext_f32(vget_low_f32(vQ1), vget_high_f32(vQ1), 1);
197 
198  vQ2yz = vext_f32(vget_low_f32(vQ2), vget_high_f32(vQ2), 1);
199  vQ2xz = vext_f32(vQ2zx, vQ2zx, 1);
200 
201  A1 = vcombine_f32(vget_low_f32(vQ1), vQ1zx); // X Y z x
202  B1 = vcombine_f32(vdup_lane_f32(vget_high_f32(vQ2), 1), vQ2wx); // W W W X
203 
204  A2 = vcombine_f32(vQ1yz, vget_low_f32(vQ1));
205  B2 = vcombine_f32(vQ2zx, vdup_lane_f32(vget_low_f32(vQ2), 1));
206 
207  A3 = vcombine_f32(vQ1zx, vQ1yz); // Z X Y Z
208  B3 = vcombine_f32(vQ2yz, vQ2xz); // Y Z x z
209 
210  A1 = vmulq_f32(A1, B1);
211  A2 = vmulq_f32(A2, B2);
212  A3 = vmulq_f32(A3, B3); // A3 *= B3
213  A0 = vmulq_lane_f32(vQ2, vget_high_f32(vQ1), 1); // A0 * B0
214 
215  A1 = vaddq_f32(A1, A2); // AB12 = AB1 + AB2
216  A0 = vsubq_f32(A0, A3); // AB03 = AB0 - AB3
217 
218  // change the sign of the last element
219  A1 = (float32x4_t)veorq_s32((int32x4_t)A1, (int32x4_t)b3vPPPM);
220  A0 = vaddq_f32(A0, A1); // AB03 + AB12
221 
222  target.set128(A0);
223 #else
224  real_t ax = this->x, ay = this->y, az = this->z, aw = this->w;
225  real_t bx = q.x, by = q.y, bz = q.z, bw = q.w;
226 
227  target.x = ax * bw + aw * bx + ay * bz - az * by;
228  target.y = ay * bw + aw * by + az * bx - ax * bz;
229  target.z = az * bw + aw * bz + ax * by - ay * bx;
230  target.w = aw * bw - ax * bx - ay * by - az * bz;
231 #endif
232  return target;
233  }
